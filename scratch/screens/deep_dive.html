def extract_entities(text: str) -> List[Entity]:
    doc = nlp(text)
    return [
        Entity(label=ent.label_, text=ent.text)
        for ent in doc.ents 
        if ent.label_ in TARGET_ENTITIES
    ]

03. Deduplication Vector Search

Active

Semantic deduplication using pgvector. Computes embeddings for incoming texts and performs cosine similarity checks against recent historical data to prevent duplicate signal processing. 

-- pgvector similarity threshold check
SELECT id, title, embedding <=> :new_embedding AS distance
FROM documents
WHERE embedding <=> :new_embedding < 0.15
ORDER BY distance ASC
LIMIT 1;

async def analyze_signal(document: Document) -> Insight:
    # 1. Fast triage via Groq
    category = await groq_client.categorize(document.text)
    
    if category.relevance_score > 0.8:
        # 2. Deep analysis via Gemini
        return await gemini_client.analyze(
            text=document.text,
            context=category.context
        )
